CREATE TABLE kurly(
  idx INT NOT NULL AUTO_INCREMENT,
  id     VARCHAR(16)   NOT NULL,    
  pw     VARCHAR(30)   NOT NULL,
  name   VARCHAR(16)   NOT NULL,
  email  VARCHAR(250)  NOT NULL,
  hp     VARCHAR(13)   NOT NULL,
  address   VARCHAR(16)   NOT NULL,
  gender VARCHAR(16),
  birth  VARCHAR(16),
  add    VARCHAR(250),
  chk    VARCHAR(250) NOT NULL,
